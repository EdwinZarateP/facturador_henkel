let
    SourceFiles = SharePoint.Files("https://bluecargogroup.sharepoint.com/sites/Almacenamiento", [ApiVersion = 15]),

    FilesExportacion =
        Table.SelectRows(
            SourceFiles,
            each Text.Contains([Folder Path], "FACTURACION")
                and [Attributes]?[Hidden]? <> true
                and Text.StartsWith([Name], "exportacion")
        ),

    AddTransformedTable =
        Table.AddColumn(
            FilesExportacion,
            "Tabla",
            each #"Transformar archivo (7)"([Content]),
            type table
        ),

    KeepOnlyNeededCols = Table.SelectColumns(AddTransformedTable, {"Name", "Tabla"}),
    ExpandTransformed =
        Table.ExpandTableColumn(
            KeepOnlyNeededCols,
            "Tabla",
            Table.ColumnNames(#"Transformar archivo (7)"(#"Archivo de ejemplo (7)"))
        ),
    #"Filas filtradas" = Table.SelectRows(ExpandTransformed, each ([Delivery] <> null)),
    SelectColumnsEarly =
        Table.SelectColumns(
            #"Filas filtradas",
            {"Name", "Delivery", "Fecha factura", "Material", "Texto breve de material", "Ctd Ent.(UMV)", "Canal distribución"}
        ),
    SetTypes =
        Table.TransformColumnTypes(
            SelectColumnsEarly,
            {
                {"Name", type text},
                {"Delivery", type text},
                {"Material", type text},
                {"Texto breve de material", type text},
                {"Ctd Ent.(UMV)", Int64.Type},
                {"Canal distribución", type text},
                {"Fecha factura", type date}
            }
        ),

    SplitFileName =
        Table.SplitColumn(
            SetTypes,
            "Name",
            Splitter.SplitTextByDelimiter("_", QuoteStyle.Csv),
            {"prefijo_archivo", "negocio_raw", "fecha_raw"}
        ),

    RemovePrefix = Table.RemoveColumns(SplitFileName, {"prefijo_archivo"}),
    CleanFechaRaw = Table.ReplaceValue(RemovePrefix, ".xlsx", "", Replacer.ReplaceText, {"fecha_raw"}),

    NormalizeNegocio =
        Table.TransformColumns(
            CleanFechaRaw,
            {{"negocio_raw", each if _ = "cons" then "CONSUMER" else if _ = "prof" then "PROFESIONAL" else _, type text}}
        ),

    SetParsedTypes = Table.TransformColumnTypes(NormalizeNegocio, {{"fecha_raw", type date}}),

    RenameToFinal =
        Table.RenameColumns(
            SetParsedTypes,
            {
                {"Delivery", "entrega"},
                {"Material", "material"},
                {"Texto breve de material", "descripcion"},
                {"Ctd Ent.(UMV)", "cantidad"},
                {"Canal distribución", "canal"},
                {"negocio_raw", "negocio"},
                {"fecha_raw", "fecha"}
            }
        ),

    JoinIDHEspeciales =
        Table.NestedJoin(
            RenameToFinal,
            {"material"},
            IDH_ESPECIALES,
            {"idh"},
            "IDH_ESPECIALES",
            JoinKind.LeftOuter
        ),

    JoinRangos =
        Table.NestedJoin(
            JoinIDHEspeciales,
            {"fecha"},
            RANGOS_FECHAS,
            {"fecha"},
            "RANGOS_FECHAS",
            JoinKind.LeftOuter
        ),
    ExpandRangos =
        Table.ExpandTableColumn(
            JoinRangos,
            "RANGOS_FECHAS",
            {"fecha_inicial", "fecha_final"},
            {"fecha_inicial", "fecha_final"}
        ),

    SetRangeTypes =
        Table.TransformColumnTypes(
            ExpandRangos,
            {
                {"fecha_inicial", type date},
                {"fecha_final", type date}
            }
        ),

    AddFlagVentanaFactura =
        Table.AddColumn(
            SetRangeTypes,
            "en_ventana_factura",
            each
                [fecha_inicial] <> null
                and [fecha_final] <> null
                and [Fecha factura] <> null
                and [Fecha factura] >= [fecha_inicial]
                and [Fecha factura] <= [fecha_final],
            type logical
        ),

    FilterVentanaFactura = Table.SelectRows(AddFlagVentanaFactura, each [en_ventana_factura] = true),

    RemoveFlag = Table.RemoveColumns(FilterVentanaFactura, {"en_ventana_factura"}),

    #"Columna duplicada" = Table.DuplicateColumn(RemoveFlag, "negocio", "negocio_facturador"),

    JoinHuellas =
        Table.NestedJoin(
            #"Columna duplicada",
            {"material", "fecha"},
            HUELLAS,
            {"idh", "fecha"},
            "HUELLAS",
            JoinKind.LeftOuter
        ),

    ExpandHuellas =
        Table.ExpandTableColumn(
            JoinHuellas,
            "HUELLAS",
            {"unidades_pallet", "unidades_caja"},
            {"unidades_pallet", "unidades_caja"}
        ),

    AddPallets =
        Table.AddColumn(
            ExpandHuellas,
            "pallets",
            each if [unidades_pallet] = null or [unidades_pallet] = 0 then null else Number.RoundUp([cantidad] / [unidades_pallet]),
            Int64.Type
        ),

    AddCajas =
        Table.AddColumn(
            AddPallets,
            "cajas",
            each if [unidades_caja] = null or [unidades_caja] = 0 then null else Number.RoundUp([cantidad] / [unidades_caja]),
            Int64.Type
        ),

    AddUnidades =
        Table.AddColumn(
            AddCajas,
            "unidades",
            each if [unidades_caja] = null or [cajas] = null then null else [unidades_caja] * [cajas],
            Int64.Type
        ),

    Grouped =
        Table.Group(
            AddUnidades,
            {"fecha", "negocio", "negocio_facturador", "canal"},
            {
                {"ALISTAMIENTO Y DESPACHO PALLETS EXPO", each List.Sum([pallets]), type number},
                {"ALISTAMIENTO Y DESPACHO CAJAS EXPO", each List.Sum([cajas]), type number},
                {"ALISTAMIENTO Y DESPACHO UND EXPO", each List.Sum([unidades]), type number}
            }
        ),

    DuplicateUND = Table.DuplicateColumn(Grouped, "ALISTAMIENTO Y DESPACHO UND EXPO", "unidades"),

    UnpivotServicios =
        Table.UnpivotOtherColumns(
            DuplicateUND,
            {"fecha", "negocio", "negocio_facturador", "canal", "unidades"},
            "atributo_servicio",
            "valor"
        ),

    AddServicio = Table.AddColumn(UnpivotServicios, "servicio", each [atributo_servicio] & " " & [canal], type text),

    RemoveServicioAux = Table.RemoveColumns(AddServicio, {"canal", "atributo_servicio"}),

    ReorderFinal = Table.ReorderColumns(RemoveServicioAux, {"fecha", "negocio", "negocio_facturador", "servicio", "valor", "unidades"}),

    AddProcesoExtendido = Table.AddColumn(ReorderFinal, "proceso_extendido", each "PICKING " & [negocio], type text),

    AddMacroProceso = Table.AddColumn(AddProcesoExtendido, "macro_proceso", each "OUT BOUND", type text),

    AddProcesoAbreviado = Table.AddColumn(AddMacroProceso, "proceso_abreviado", each "OUB", type text),

    AddTabla = Table.AddColumn(AddProcesoAbreviado, "tabla", each "EXPORTACIONES", type text)

in
    AddTabla
